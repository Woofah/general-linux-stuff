#FSTAB code for adding extra harddrives for automount
#I'm sure I'll be updating this a lot when I have to do reformats or get new hard drives

#Commands to run first
sudo blkid #Copy UUIDs
mkdir ~/Storage1TB
mkdir ~/Storage2TB
mkdir ~/Storage4TB
sudo mkdir /etc/backup
sudo cp /etc/fstab /etc/backup/fstab
sudo nano /etc/fstab

#Restore FSTAB
cd /etc/
sudo rm fstab
sudo cp /etc/backup/fstab /etc/

#Commands to allow full access to drives
sudo chown -v ericparsley:ericparsley /media/sdb1
sudo chown -v ericparsley:ericparsley /media/sdc1
sudo chown -v ericparsley:ericparsley /media/sdd1
