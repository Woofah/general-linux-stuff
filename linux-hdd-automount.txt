#FSTAB code for adding extra harddrives for automount
#I'm sure I'll be updating this a lot when I have to do reformats or get new hard drives

#Commands to run first
sudo blkid #Copy UUIDs
mkdir ~/Storage1
mkdir ~/Storage2
mkdir ~/Storage3
sudo mkdir /etc/backup
sudo cp /etc/fstab /etc/backup/fstab
sudo nano /etc/fstab

#Add to fstab file
#dev/sdb1
UUID=b544c215-80a9-489d-a917-45d20f1b334b       /home/ericparsley/Storage1      ext4       defaults,noatime       0       2

#dev/sdc1
UUID=8c2b5414-175e-4a7e-9208-a1366f057328       /home/ericparsley/Storage2      ext4       defaults,noatime       0       2

#dev/sdd1
UUID=3001a4f4-8a5c-4f88-b980-bf130c2dd42f       /home/ericparsley/Storage3      ext4       defaults,noatime       0       2

#Commands to allow full access to drives
sudo chown -v ericparsley:ericparsley /home/ericparsley/Storage1
sudo chown -v ericparsley:ericparsley /home/ericparsley/Storage2
sudo chown -v ericparsley:ericparsley /home/ericparsley/Storage3

#Restore fstab if needed
cd /etc/
sudo rm fstab
sudo cp /etc/backup/fstab /etc/
