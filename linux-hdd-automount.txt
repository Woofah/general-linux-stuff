#fstab code for adding extra harddrives for automount
#I'm sure I'll be updating this a lot when I have to do reformats or get new hard drives

#Commands to run first
sudo blkid #Copy UUIDs
mkdir ~/CloudStorage && mkdir ~/GameStorage && mkdir ~/OtherStorage
sudo mkdir /etc/backup && sudo cp /etc/fstab /etc/backup/fstab
sudo nano /etc/fstab

#Add to fstab file
#dev/sdb1
UUID=07ca7b03-74ad-4155-ad9b-8786d3cc8a57       /home/ericparsley/CloudStorage      ext4       defaults,noatime       0       2

#dev/sdc1
UUID=cf883070-cdc8-4edb-ada6-9b533b839ebe       /home/ericparsley/GameStorage      ext4       defaults,noatime       0       2

#dev/sdd1
UUID=7bcab2d7-b501-4148-bdc3-f9b0ba4007e7       /home/ericparsley/OtherStorage      ext4       defaults,noatime       0       2

#Commands to allow full access to drives
sudo chown -v ericparsley:ericparsley /home/ericparsley/CloudStorage && sudo chown -v ericparsley:ericparsley /home/ericparsley/GameStorage && sudo chown -v ericparsley:ericparsley /home/ericparsley/OtherStorage

#Restore fstab if needed
cd /etc/ && sudo rm fstab && sudo cp /etc/backup/fstab /etc/
